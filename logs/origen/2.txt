import { execSync } from "child_process"
import fs from "fs"
import path from "path"

export function processVoices(timeline, outputDir) {
  const assets = []
  const subtitles = []

  timeline
    .filter(e => e.kind === "voice")
    .forEach((event, i) => {
      const audioPath = path.join(outputDir, `voice_${i}.wav`)
      const metaPath = path.join(outputDir, `voice_${i}.json`)

      execSync(
        `edge-tts --voice "${event.voice}" \
        --rate "${event.rate}" \
        --pitch "${event.pitch}" \
        --text "${event.text}" \
        --write-media "${audioPath}" \
        --write-subtitles "${metaPath}"`
      )

      const meta = JSON.parse(fs.readFileSync(metaPath, "utf-8"))

      assets.push({
        kind: "audio",
        src: audioPath,
        start: event.start,
        duration: meta.duration
      })

      if (event.subtitles) {
        meta.cues.forEach(cue => {
          subtitles.push({
            kind: "text",
            start: event.start + cue.start,
            duration: cue.duration,
            content: cue.text
          })
        })
      }
    })

  return { assets, subtitles }
}

// engine/exporters/capcut_exporter.js
import fs from "fs";
import path from "path";

export function exportToCapCut(ritual, outDir) {

    const clips = ritual.timeline.map((c, i) => ({
        id: i,
        type: c.clipType ?? c.type,
        src: c.src ?? null,
        text: c.value ?? null,
        start: c.start,
        duration: c.duration ?? 0,
        layer: c.layer ?? 0,
        style: c.style ?? {}
    }));

    const project = {
        name: ritual.name,
        fps: 30,
        resolution: "1080x1920",
        tracks: {
            video: clips.filter(c => c.type !== "audio"),
            audio: clips.filter(c => c.type === "audio")
        }
    };

    const file = path.join(
        outDir,
        ${ritual.name}.capcut.json
    );

    fs.writeFileSync(file, JSON.stringify(project, null, 2), "utf-8");
    console.log("🎬 CapCut exportado en:\n" + file);
}
